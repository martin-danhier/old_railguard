11  // Type used to represent a full Entity id.
12  //
13  // A typedef is used to allow better maintability:
14  // If the size of the integer type needs to be increased in the future,

16  typedef uint64_t eid_t;
17  // Type used to represent the unique part of an Entity id.
18  // It needs to be large enough to fit in every possible unique id,
19  // but as small as possible to avoid taking not needed space
20  typedef uint8_t unique_eid_t;
21 
22  // Number of bits used for the "unique" part of an entity id
23  //
24  // The "unique part" is used to differentiate entities created at the same index
25  // The number of possible unique ids defines the maximum number of entities that can be created in a slot
26  // without looping.
27  // Note: these are the ids that will be stored in the table. That means that the smaller the unique part is,
28  // the less space the entity table will take, and the more ids will fit in the cache.
29  //
30  // Formula: 2^UNIQUE_BITS
31  constexpr uint8_t UNIQUE_BITS = 8;
32  // Number of bits used for the "index" part of an entity id
33  //
34  // The "index part" is used to index an id in the lookup table.
35  // The number of possible indexes defines the maximum number of simultaneous alive entities.
36  //
37  // Formula: 2^INDEX_BITS
38  constexpr uint8_t INDEX_BITS = 56;
39 
40  // Static assert to prevent mis configuration of the above parameters
41  // They must always sum up to the the size of the entity id
42  // If in the future, the id must be divided in more than 2 parts, this assertion will need to be
43  // updated.
44  static_assert(
45  static_cast<size_t>(UNIQUE_BITS + INDEX_BITS) == sizeof(eid_t) * 8,
46  "The unique and index bits must sum up to the length of the type used to represent Entity ids.");
47  // Check if the unique_eid_t type can hold every possible unique id
48  static_assert(
49  UNIQUE_BITS <= sizeof(unique_eid_t) * 8,
50  "The unique_eid_t type must be large enough to fit in every possible unique part of an Entity id.");
